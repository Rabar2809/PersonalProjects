class Nutrition extends StatefulWidget {               // Fourth Page
  
  State<StatefulWidget> createState() {
    return _nutrition();
  }
  
}

class _nutrition extends State<Nutrition>{        // fourth page by having a stateful widget and slider to scroll each diet

  static final GlobalKey<ScaffoldState> _ScaffoldKey = GlobalKey<ScaffoldState>();

  final controll = PageController(
      initialPage: 1
  );

  @override
  Widget build(BuildContext context){

    // ignore: unused_local_variable
    Size _screenSize = MediaQuery.of(context).size;

    return Scaffold(
      key: _ScaffoldKey,
      appBar: AppBar(
        centerTitle: true,
        backgroundColor: Colors.grey,
        title: const Text('Cards that can be scrolled', style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold ),),

      ),

      body: PageView(
        controller: controll,
        scrollDirection: Axis.horizontal,
        

        physics: const BouncingScrollPhysics(),
        pageSnapping: true,
        children: <Widget>[

          GestureDetector(                        // a gestureDetector that has a card inside. Same applies for the other two
            onTap: () => setState(() {
                      var snackBar = const SnackBar(content: Text('Card is pressed'));
                            ScaffoldMessenger.of(context).showSnackBar(snackBar);
            }),
            child: Container(
            color: Colors.grey,
            child: Card(
              color: Colors.blueGrey,
              elevation: 4,
              margin: const EdgeInsets.all(15),
              semanticContainer: true,
              child: cardPic(
                images: "f1.jpg",
                
              ),
            ),

          ),
          ),

          GestureDetector(                      // second card inside the gesture detector
            onTap: () => setState(() {
                  var snackBar = const SnackBar(content: Text('Card is pressed'));
                            ScaffoldMessenger.of(context).showSnackBar(snackBar);
            }),
            child: Container(
            color: Colors.grey,
            child: Card(
              color: Colors.blueGrey,
              elevation: 4,
              margin: const EdgeInsets.all(15),
              semanticContainer: true,
              child: cardPic(
                images: "f2.jpg",
                
              ),
            ),

          ),
          ),

          GestureDetector(                              // third card inside gesture detector
            onTap: () => setState(() {
                var snackBar = const SnackBar(content: Text('Card is pressed'));
                            ScaffoldMessenger.of(context).showSnackBar(snackBar);
            }),
            child: Container(
            color: Colors.grey,
            child: Card(
              color: Colors.blueGrey,
              elevation: 4,
              margin: const EdgeInsets.all(15),
              semanticContainer: true,
